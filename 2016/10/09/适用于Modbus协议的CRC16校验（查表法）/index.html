uint   N_CRC16(uchar *updata, uint len)
{
    uchar uchCRCHi = 0xff;
    uchar uchCRCLo = 0xff;
    uint  uindex;
    while (len--)
    {
        uindex = uchCRCHi^*updata++;
        uchCRCHi = uchCRCLo^auchCRCHi[uindex];
        uchCRCLo = auchCRCLo[uindex];
    }
    return (uchCRCHi << 8 | uchCRCLo);
}
